DELIMITER $$

CREATE PROCEDURE insertAccumulatedAmount (
    IN p_user_id VARCHAR(15),
    IN p_pos_id INT(11),
    IN p_payment_date DATE
)
BEGIN
    INSERT INTO accumulated_amount_tbl (
        user_id,
        pos_id,
        payment_date,
        begin_or_no,
        end_or_no,
        no_of_transaction,
        no_of_void,
	total_sales,
        discount,
        vatable,
        vat_exempt,
        vat,
        zero_rated,
        z_counter,
        begin_void_no,
        end_void_no,
        void_amount,
        grand_total,
        printed,
        upload
    )
    SELECT 
        p_user_id,
        p_pos_id,
        p_payment_date,
        MIN(or_no),
        MAX(or_no),
        COUNT(or_no),
        0,
        SUM(total_amount),
        SUM(discount),
        SUM(vatable),
        SUM(vat_exempt),
        SUM(vat),
        SUM(zero_rated),
        0,
        0,
        0,
        '0.00',
        SUM(total_amount),
        'no',
        'no'
    FROM or_tbl
    WHERE cashier_id = p_user_id
      AND payment_date = p_payment_date;
END $$

DELIMITER ;
