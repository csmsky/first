DELIMITER $$

CREATE PROCEDURE insertCashFloat (
    IN  p_date_cashfloat  DATE,
    IN  p_time_cashfloat  TIME,
    IN  p_pos_id       INT,
    IN  p_cashier_id   VARCHAR(15),
    IN  p_cash_float   DECIMAL(15,2),
    IN  p_approvedby   VARCHAR(15),
    IN  p_reason       VARCHAR(50),
    OUT p_message      VARCHAR(255)
)
BEGIN
    DECLARE v_exists INT DEFAULT 0;

    -- Check if accumulated record exists
    SELECT COUNT(*)
    INTO v_exists
    FROM accumulated_amount_tbl
    WHERE pos_id = p_pos_id
      AND user_id = p_cashier_id
      AND payment_date = p_date_cashfloat;

    START TRANSACTION;

    IF v_exists = 0 THEN
        -- ❗ No accumulated record = opening not yet created
        SET p_message = 'NO ACCUMULATED RECORD FOUND';
        ROLLBACK;
    ELSE
        -- Update accumulated table
        UPDATE accumulated_amount_tbl
        SET cash_float = IFNULL(cash_float, 0) + p_cash_float
        WHERE pos_id = p_pos_id
          AND user_id = p_cashier_id
          AND payment_date = p_date_cashfloat;

        -- Insert audit trail (ALWAYS INSERT)
        INSERT INTO cash_float_tbl (
            cashier_id,
            cashfloat_amount,
            reason,
            approvedby,
            date,
	    time
        ) VALUES (
            p_cashier_id,
            p_cash_float,
            p_reason,
            p_approvedby,
            NOW(),
	    CURTIME()	
        );

        COMMIT;
        SET p_message = 'CASH FLOAT SAVED';
    END IF;

END$$
DELIMITER ;
